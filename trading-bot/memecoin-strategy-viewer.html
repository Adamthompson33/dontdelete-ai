function generateRecommendation(analysis) {
  const { signal, holders, authority, rugScore, sentiment } = analysis;

  // === HARD REJECTIONS ===
  
  if (rugScore > 60) {
    return { action: 'REJECT', reason: `üö´ High rug risk (score: ${rugScore})` };
  }
  
  if (authority?.hasMintAuthority) {
    return { action: 'REJECT', reason: 'üö´ Mint authority enabled' };
  }
  
  if (sentiment?.botActivity === 'high') {
    return { action: 'REJECT', reason: 'üö´ High bot activity on X' };
  }
  
  if (sentiment && !sentiment.organic) {
    return { action: 'REJECT', reason: 'üö´ Not organic movement' };
  }

  // === CALCULATE CONFIDENCE ===
  
  let confidence = signal.signalStrength;
  
  if (rugScore < 20) confidence += 0.15;
  if (holders?.holderCount > 500) confidence += 0.1;
  if (sentiment?.sentimentScore >= 8) confidence += 0.2;
  if (sentiment?.organic) confidence += 0.1;
  if (sentiment?.influencerMentions) confidence += 0.15;

  // Position sizing based on confidence
  const positionSol = config.risk.maxPositionSol * confidence;

  if (confidence >= 0.6) {
    return { action: 'BUY', confidence, positionSol };
  }
  
  return { action: 'WATCH', confidence };
}
